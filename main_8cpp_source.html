130/**
131 * @brief Arduino main loop function, animates, renders, and updates the selected project.
132 *
133 * If PROJECT_VERIFY_HARDWARE is defined, this function is disabled.
134 */
135void loop() {
136 #ifndef PROJECT_VERIFY_HARDWARE
137 bluetooth.run();
138
139 BTcolor = bluetooth.getColor();
140
141 if(bluetooth.getMenuButton()) { // faking the menu button
142 digitalWrite(23, LOW);
143 } else {
144 digitalWrite(23, HIGH);
145 }
146
147 if ( boopSensor.readProximity(proximity_data) && menu.UseBoopSensor() ) { // proximity sensor custom implementation
148 if (proximity_data > 18) { // distance for triggering boop
149 booped = true;
150 } else {
151 booped = false;
152 }
153 } else if (!menu.UseBoopSensor()) {
154 booped = false;
155 }
156
157 if(!(BTcolor.R == 0 && BTcolor.G == 0 && BTcolor.B == 0)) {
158 color = BTcolor;
159 }
160
161 if (bluetooth.getMenuOverride()) { // check if menu gets overriden by phone
162 if(bluetooth.getGayMode()) {
163 project.SelectColor(9);
164 } else {
165 if (bluetooth.getExpression() != 8 && !booped) {
166 project.CustomFaceColor(color);
167 } else if (bluetooth.getExpression() == 8 && !bluetooth.getExpressionOverride() && !booped) {
168 project.CustomFaceColor(color);
169 }
170 }
171 }
172
173 if (bluetooth.getExpressionOverride()) { // if both menu and bluetooth sets expression they will mix
174 if(!booped) {
175 project.SelectFace(bluetooth.getExpression());
176 }
177 }
178 if(booped) {
179 project.SelectFace(8);
180 }
181
182 float ratio = (float)(millis() % 5000) / 5000.0f; ///< Calculates animation ratio based on time.
183
184 project.Animate(ratio); ///< Animates the project based on the current ratio.
185 project.Render(); ///< Renders the project's scene.
186 project.Display(); ///< Displays the rendered frame.
187 project.PrintStats(); ///< Outputs debugging and performance statistics.
188 #endif
189}
